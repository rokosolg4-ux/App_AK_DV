const express = require('express'); const { Pool } = require('pg'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const pool = new Pool({ user: 'user', host: 'localhost', database: 'avtokraski_db', password: 'password', port: 5432, }); // 1. ОТДАЕМ КАТЕГОРИИ app.get('/api/categories', async (req, res) => { try { const result = await pool.query('SELECT * FROM categories ORDER BY id'); res.json(result.rows); } catch (err) { res.status(500).json({ error: err.message }); } }); // 2. ОТДАЕМ ТОВАРЫ (С ФИЛЬТРАЦИЕЙ!) app.get('/api/stock', async (req, res) => { const { category } = req.query; // Получаем ID папки от фронтенда try { // Начало запроса let queryText = ` SELECT p.id, p.name AS product_name, c.sku, pr.price, s.quantity as qty, p.category_id FROM products p LEFT JOIN characteristics c ON p.id = c.product_id LEFT JOIN reg_prices pr ON c.id = pr.characteristic_id LEFT JOIN reg_stock_balance s ON c.id = s.characteristic_id `; // ВАЖНЫЙ МОМЕНТ: Если пришел ID категории, добавляем фильтр const queryParams = []; if (category) { queryText += ` WHERE p.category_id = $1`; queryParams.push(category); } // Сортировка по имени, чтобы было красиво queryText += ` ORDER BY p.name ASC`; const result = await pool.query(queryText, queryParams); res.json(result.rows); } catch (err) { console.error(err); res.status(500).json({ error: "Ошибка базы данных" }); } }); app.listen(3000, () => console.log('Server running on port 3000'));
